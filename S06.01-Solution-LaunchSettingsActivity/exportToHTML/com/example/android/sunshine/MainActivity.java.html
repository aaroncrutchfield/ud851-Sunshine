57       @Override 
58       protected void onCreate(Bundle savedInstanceState) { 
59           super.onCreate(savedInstanceState); 
60           setContentView(R.layout.activity_forecast);

62           /* 
63            * Using findViewById, we get a reference to our RecyclerView from xml. This allows us to 
64            * do things like set the adapter of the RecyclerView and toggle the visibility.

66           mRecyclerView = (RecyclerView) findViewById(R.id.recyclerview_forecast); 
67    
68           /* This TextView is used to display errors and will be hidden if there are no errors */ 
69           mErrorMessageDisplay = (TextView) findViewById(R.id.tv_error_message_display);

71           /* 
72            * A LinearLayoutManager is responsible for measuring and positioning item views within a 
73            * RecyclerView into a linear list. This means that it can produce either a horizontal or 
74            * vertical list depending on which parameter you pass in to the LinearLayoutManager 
75            * constructor. In our case, we want a vertical list, so we pass in the constant from the 
76            * LinearLayoutManager class for vertical lists, LinearLayoutManager.VERTICAL. 
77            * 
78            * There are other LayoutManagers available to display your data in uniform grids, 
79            * staggered grids, and more! See the developer documentation for more details. 
80            */ 
81           int recyclerViewOrientation = LinearLayoutManager.VERTICAL; 
82    
83           /* 
84            *  This value should be true if you want to reverse your layout. Generally, this is only 
85            *  true with horizontal lists that need to support a right-to-left layout. 
86            */ 
87           boolean shouldReverseLayout = false; 
88           LinearLayoutManager layoutManager 
89                   = new LinearLayoutManager(this, recyclerViewOrientation, shouldReverseLayout); 
90           mRecyclerView.setLayoutManager(layoutManager); 
91    
92           /* 
93            * Use this setting to improve performance if you know that changes in content do not 
94            * change the child layout size in the RecyclerView 
95            */ 
96           mRecyclerView.setHasFixedSize(true); 
97    
98           /* 
99            * The ForecastAdapter is responsible for linking our weather data with the Views that 
100           * will end up displaying our weather data. 
101           */ 
102          mForecastAdapter = new ForecastAdapter(this); 
103   
104          /* Setting the adapter attaches it to the RecyclerView in our layout. */ 
105          mRecyclerView.setAdapter(mForecastAdapter); 
106   
107          /* 
108           * The ProgressBar that will indicate to the user that we are loading data. It will be 
109           * hidden when no data is loading. 
110           * 
111           * Please note: This so called "ProgressBar" isn't a bar by default. It is more of a 
112           * circle. We didn't make the rules (or the names of Views), we just follow them. 
113           */ 
114          mLoadingIndicator = (ProgressBar) findViewById(R.id.pb_loading_indicator); 
115   
116          /* 
117           * This ID will uniquely identify the Loader. We can use it, for example, to get a handle 
118           * on our Loader at a later point in time through the support LoaderManager. 
119           */ 
120          int loaderId = FORECAST_LOADER_ID; 
121   
122          /* 
123           * From MainActivity, we have implemented the LoaderCallbacks interface with the type of 
124           * String array. (implements LoaderCallbacks<String[]>) The variable callback is passed 
125           * to the call to initLoader below. This means that whenever the loaderManager has 
126           * something to notify us of, it will do so through this callback. 
127           */ 
128          LoaderCallbacks<String[]> callback = MainActivity.this; 
129   
130          /* 
131           * The second parameter of the initLoader method below is a Bundle. Optionally, you can 
132           * pass a Bundle to initLoader that you can then access from within the onCreateLoader 
133           * callback. In our case, we don't actually use the Bundle, but it's here in case we wanted 
134           * to. 
135           */ 
136          Bundle bundleForLoader = null; 
137   
138          /* 
139           * Ensures a loader is initialized and active. If the loader doesn't already exist, one is 
140           * created and (if the activity/fragment is currently started) starts the loader. Otherwise 
141           * the last created loader is re-used. 
142           */ 
143          getSupportLoaderManager().initLoader(loaderId, bundleForLoader, callback); 
144   
145          Log.d(TAG, "onCreate: registering preference changed listener"); 
146      }

148      /** 
149       * Instantiate and return a new Loader for the given ID. 
150       * 
151       * @param id The ID whose loader is to be created. 
152       * @param loaderArgs Any arguments supplied by the caller. 
153       * 
154       * @return Return a new Loader instance that is ready to start loading. 
155       */ 
156      @Override 
157      public Loader<String[]> onCreateLoader(int id, final Bundle loaderArgs) { 
158   
159          return new AsyncTaskLoader<String[]>(this) { 
160   
161              /* This String array will hold and help cache our weather data */ 
162              String[] mWeatherData = null; 
163   
164              /** 
165               * Subclasses of AsyncTaskLoader must implement this to take care of loading their data. 
166               */ 
167              @Override 
168              protected void onStartLoading() { 
169                  if (mWeatherData != null) { 
170                      deliverResult(mWeatherData); 
171                  } else { 
172                      mLoadingIndicator.setVisibility(View.VISIBLE); 
173                      forceLoad(); 
174                  } 
175              } 
176   
177              /** 
178               * This is the method of the AsyncTaskLoader that will load and parse the JSON data 
179               * from OpenWeatherMap in the background. 
180               * 
181               * @return Weather data from OpenWeatherMap as an array of Strings. 
182               *         null if an error occurs 
183               */ 
184              @Override 
185              public String[] loadInBackground() { 
186   
187                  String locationQuery = SunshinePreferences 
188                          .getPreferredWeatherLocation(MainActivity.this); 
189   
190                  URL weatherRequestUrl = NetworkUtils.buildUrl(locationQuery); 
191   
192                  try { 
193                      String jsonWeatherResponse = NetworkUtils 
194                              .getResponseFromHttpUrl(weatherRequestUrl); 
195   
196                      String[] simpleJsonWeatherData = OpenWeatherJsonUtils 
197                              .getSimpleWeatherStringsFromJson(MainActivity.this, jsonWeatherResponse); 
198   
199                      return simpleJsonWeatherData; 
200                  } catch (Exception e) { 
201                      e.printStackTrace(); 
202                      return null; 
203                  } 
204              } 
205   
206              /** 
207               * Sends the result of the load to the registered listener. 
208               * 
209               * @param data The result of the load 
210               */ 
211              public void deliverResult(String[] data) { 
212                  mWeatherData = data; 
213                  super.deliverResult(data); 
214              } 
215          }; 
216      } 
217   
218      /** 
219       * Called when a previously created loader has finished its load. 
220       * 
221       * @param loader The Loader that has finished. 
222       * @param data The data generated by the Loader. 
223       */ 
224      @Override 
225      public void onLoadFinished(Loader<String[]> loader, String[] data) { 
226          mLoadingIndicator.setVisibility(View.INVISIBLE); 
227          mForecastAdapter.setWeatherData(data); 
228          if (null == data) { 
229              showErrorMessage(); 
230          } else { 
231              showWeatherDataView(); 
232          } 
233      }

242      @Override 
243      public void onLoaderReset(Loader<String[]> loader) { 
244          /* 
245           * We aren't using this method in our example application, but we are required to Override 
246           * it to implement the LoaderCallbacks<String> interface 
247           */ 
248      }

250      /** 
251       * This method is used when we are resetting data, so that at one point in time during a 
252       * refresh of our data, you can see that there is no data showing. 
253       */ 
254      private void invalidateData() { 
255          mForecastAdapter.setWeatherData(null); 
256      } 
257   
258      /** 
259       * This method uses the URI scheme for showing a location found on a map in conjunction with 
260       * an implicit Intent. This super-handy intent is detailed in the "Common Intents" page of 
261       * Android's developer site: 
262       * 
263       * @see "http://developer.android.com/guide/components/intents-common.html#Maps" 
264       * <p> 
265       * Protip: Hold Command on Mac or Control on Windows and click that link to automagically 
266       * open the Common Intents page 
267       */ 
268      private void openLocationInMap() { 
269          String addressString = "1600 Ampitheatre Parkway, CA"; 
270          Uri geoLocation = Uri.parse("geo:0,0?q=" + addressString); 
271   
272          Intent intent = new Intent(Intent.ACTION_VIEW); 
273          intent.setData(geoLocation); 
274   
275          if (intent.resolveActivity(getPackageManager()) != null) { 
276              startActivity(intent); 
277          } else { 
278              Log.d(TAG, "Couldn't call " + geoLocation.toString() + ", no receiving apps installed!"); 
279          } 
280      }

282      /** 
283       * This method is for responding to clicks from our list. 
284       * 
285       * @param weatherForDay String describing weather details for a particular day 
286       */ 
287      @Override 
288      public void onClick(String weatherForDay) { 
289          Context context = this; 
290          Class destinationClass = DetailActivity.class; 
291          Intent intentToStartDetailActivity = new Intent(context, destinationClass); 
292          intentToStartDetailActivity.putExtra(Intent.EXTRA_TEXT, weatherForDay); 
293          startActivity(intentToStartDetailActivity); 
294      } 
295   
296      /** 
297       * This method will make the View for the weather data visible and 
298       * hide the error message. 
299       * <p> 
300       * Since it is okay to redundantly set the visibility of a View, we don't 
301       * need to check whether each view is currently visible or invisible. 
302       */ 
303      private void showWeatherDataView() { 
304          /* First, make sure the error is invisible */ 
305          mErrorMessageDisplay.setVisibility(View.INVISIBLE); 
306          /* Then, make sure the weather data is visible */ 
307          mRecyclerView.setVisibility(View.VISIBLE); 
308      } 
309   
310      /** 
311       * This method will make the error message visible and hide the weather 
312       * View. 
313       * <p> 
314       * Since it is okay to redundantly set the visibility of a View, we don't 
315       * need to check whether each view is currently visible or invisible. 
316       */ 
317      private void showErrorMessage() { 
318          /* First, hide the currently visible data */ 
319          mRecyclerView.setVisibility(View.INVISIBLE); 
320          /* Then, show the error */ 
321          mErrorMessageDisplay.setVisibility(View.VISIBLE); 
322      }

334      @Override 
335      public boolean onOptionsItemSelected(MenuItem item) { 
336          int id = item.getItemId(); 
337   
338          if (id == R.id.action_refresh) { 
339              invalidateData(); 
340              getSupportLoaderManager().restartLoader(FORECAST_LOADER_ID, null, this); 
341              return true; 
342          } 
343   
344          if (id == R.id.action_map) { 
345              openLocationInMap(); 
346              return true; 
347          } 
348   
349          // COMPLETED (6) Launch SettingsActivity when the Settings option is clicked 
350          if (id == R.id.action_settings) { 
351              Intent startSettingsActivity = new Intent(this, SettingsActivity.class); 
352              startActivity(startSettingsActivity); 
353              return true; 
354          } 
355   
356          return super.onOptionsItemSelected(item); 
357      }
