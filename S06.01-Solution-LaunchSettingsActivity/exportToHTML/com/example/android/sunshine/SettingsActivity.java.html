16   package com.example.android.sunshine; 
17    
18   import android.os.Bundle; 
19   import android.support.v7.app.AppCompatActivity; 
20   import android.view.MenuItem; 
21    
22   // COMPLETED (1) Add new Activity called SettingsActivity using Android Studio wizard 
23   /** 
24    * Loads the SettingsFragment and handles the proper behavior of the up button. 
25    */ 
26   public class SettingsActivity extends AppCompatActivity { 
27    
28       @Override 
29       protected void onCreate(Bundle savedInstanceState) { 
30           super.onCreate(savedInstanceState); 
31           this.setContentView(R.layout.activity_settings); 
32           // COMPLETED (2) Set setDisplayHomeAsUpEnabled to true on the support ActionBar 
33           this.getSupportActionBar().setDisplayHomeAsUpEnabled(true); 
34       } 
35    
36       @Override 
37       public boolean onOptionsItemSelected(MenuItem item) { 
38           int id = item.getItemId(); 
39           if (id == android.R.id.home) { 
40               onBackPressed(); 
41           } 
42    
43           return super.onOptionsItemSelected(item); 
44       } 
45   }
