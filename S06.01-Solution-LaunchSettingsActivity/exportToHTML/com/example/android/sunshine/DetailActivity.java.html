1    package com.example.android.sunshine; 
2     
3    import android.content.Intent; 
4    import android.os.Bundle; 
5    import android.support.v4.app.ShareCompat; 
6    import android.support.v7.app.AppCompatActivity; 
7    import android.view.Menu; 
8    import android.view.MenuItem; 
9    import android.widget.TextView; 
10    
11   public class DetailActivity extends AppCompatActivity { 
12    
13       private static final String FORECAST_SHARE_HASHTAG = " #SunshineApp"; 
14    
15       private String mForecast; 
16       private TextView mWeatherDisplay; 
17    
18       @Override 
19       protected void onCreate(Bundle savedInstanceState) { 
20           super.onCreate(savedInstanceState); 
21           setContentView(R.layout.activity_detail); 
22    
23           mWeatherDisplay = (TextView) findViewById(R.id.tv_display_weather); 
24    
25           Intent intentThatStartedThisActivity = getIntent(); 
26    
27           if (intentThatStartedThisActivity != null) { 
28               if (intentThatStartedThisActivity.hasExtra(Intent.EXTRA_TEXT)) { 
29                   mForecast = intentThatStartedThisActivity.getStringExtra(Intent.EXTRA_TEXT); 
30                   mWeatherDisplay.setText(mForecast); 
31               } 
32           } 
33       } 
34    
35       /** 
36        * Uses the ShareCompat Intent builder to create our Forecast intent for sharing. We set the 
37        * type of content that we are sharing (just regular text), the text itself, and we return the 
38        * newly created Intent. 
39        * 
40        * @return The Intent to use to start our share. 
41        */ 
42       private Intent createShareForecastIntent() { 
43           Intent shareIntent = ShareCompat.IntentBuilder.from(this) 
44                   .setType("text/plain") 
45                   .setText(mForecast + FORECAST_SHARE_HASHTAG) 
46                   .getIntent(); 
47           return shareIntent; 
48       } 
49    
50       @Override 
51       public boolean onCreateOptionsMenu(Menu menu) { 
52           getMenuInflater().inflate(R.menu.detail, menu); 
53           MenuItem menuItem = menu.findItem(R.id.action_share); 
54           menuItem.setIntent(createShareForecastIntent()); 
55           return true; 
56       } 
57    
58       @Override 
59       public boolean onOptionsItemSelected(MenuItem item) { 
60           int id = item.getItemId(); 
61    
62           // COMPLETED (7) Launch SettingsActivity when the Settings option is clicked 
63           if (id == R.id.action_settings) { 
64               Intent startSettingsActivity = new Intent(this, SettingsActivity.class); 
65               startActivity(startSettingsActivity); 
66               return true; 
67           } 
68    
69           return super.onOptionsItemSelected(item); 
70       } 
71   }
