12   /** 
13    * The SettingsFragment serves as the display for all of the user's settings. In Sunshine, the 
14    * user will be able to change their preference for units of measurement from metric to imperial, 
15    * set their preferred weather location, and indicate whether or not they'd like to see 
16    * notifications. 
17    * 
18    * Please note: If you are using our dummy weather services, the location returned will always be 
19    * Mountain View, California. 
20    */ 
21   // COMPLETED (4) Create SettingsFragment and extend PreferenceFragmentCompat 
22   public class SettingsFragment extends PreferenceFragmentCompat implements 
23           // COMPLETED (10) Implement OnSharedPreferenceChangeListener from SettingsFragment 
24           SharedPreferences.OnSharedPreferenceChangeListener { 
25    
26       // COMPLETED (8) Create a method called setPreferenceSummary that accepts a Preference and an Object and sets the summary of the preference 
27       private void setPreferenceSummary(Preference preference, Object value) { 
28           String stringValue = value.toString(); 
29           String key = preference.getKey(); 
30    
31           if (preference instanceof ListPreference) { 
32               /* For list preferences, look up the correct display value in */ 
33               /* the preference's 'entries' list (since they have separate labels/values). */ 
34               ListPreference listPreference = (ListPreference) preference; 
35               int prefIndex = listPreference.findIndexOfValue(stringValue); 
36               if (prefIndex >= 0) { 
37                   preference.setSummary(listPreference.getEntries()[prefIndex]); 
38               } 
39           } else { 
40               // For other preferences, set the summary to the value's simple string representation. 
41               preference.setSummary(stringValue); 
42           } 
43       } 
44    
45       // COMPLETED (5) Override onCreatePreferences and add the preference xml file using addPreferencesFromResource 
46       @Override 
47       public void onCreatePreferences(Bundle bundle, String s) { 
48           /* Add 'general' preferences, defined in the XML file */ 
49           addPreferencesFromResource(R.xml.pref_general); 
50    
51           // COMPLETED (9) Set the preference summary on each preference that isn't a CheckBoxPreference 
52           SharedPreferences sharedPreferences = getPreferenceScreen().getSharedPreferences(); 
53           PreferenceScreen prefScreen = getPreferenceScreen(); 
54           int count = prefScreen.getPreferenceCount(); 
55           for (int i = 0; i < count; i++) { 
56               Preference p = prefScreen.getPreference(i); 
57               if (!(p instanceof CheckBoxPreference)) { 
58                   String value = sharedPreferences.getString(p.getKey(), ""); 
59                   setPreferenceSummary(p, value); 
60               } 
61           } 
62       } 
63    
64       // COMPLETED (13) Unregister SettingsFragment (this) as a SharedPreferenceChangedListener in onStop 
65       @Override 
66       public void onStop() { 
67           super.onStop(); 
68           /* Unregister the preference change listener */ 
69           getPreferenceScreen().getSharedPreferences() 
70                   .unregisterOnSharedPreferenceChangeListener(this); 
71       } 
72    
73       // COMPLETED (12) Register SettingsFragment (this) as a SharedPreferenceChangedListener in onStart 
74       @Override 
75       public void onStart() { 
76           super.onStart(); 
77           /* Register the preference change listener */ 
78           getPreferenceScreen().getSharedPreferences() 
79                   .registerOnSharedPreferenceChangeListener(this); 
80       } 
81    
82       // COMPLETED (11) Override onSharedPreferenceChanged to update non CheckBoxPreferences when they are changed 
83       @Override 
84       public void onSharedPreferenceChanged(SharedPreferences sharedPreferences, String key) { 
85           Preference preference = findPreference(key); 
86           if (null != preference) { 
87               if (!(preference instanceof CheckBoxPreference)) { 
88                   setPreferenceSummary(preference, sharedPreferences.getString(key, "")); 
89               } 
90           } 
91       }
