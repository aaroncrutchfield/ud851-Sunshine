29   /** 
30    * This class serves as the ContentProvider for all of Sunshine's data. This class allows us to 
31    * bulkInsert data, query data, and delete data. 
32    * <p> 
33    * Although ContentProvider implementation requires the implementation of additional methods to 
34    * perform single inserts, updates, and the ability to get the type of the data from a URI. 
35    * However, here, they are not implemented for the sake of brevity and simplicity. If you would 
36    * like, you may implement them on your own. However, we are not going to be teaching how to do 
37    * so in this course. 
38    */ 
39   public class WeatherProvider extends ContentProvider {

41       /* 
42        * These constant will be used to match URIs with the data they are looking for. We will take 
43        * advantage of the UriMatcher class to make that matching MUCH easier than doing something 
44        * ourselves, such as using regular expressions. 
45        */ 
46       public static final int CODE_WEATHER = 100; 
47       public static final int CODE_WEATHER_WITH_DATE = 101; 
48    
49       /* 
50        * The URI Matcher used by this content provider. The leading "s" in this variable name 
51        * signifies that this UriMatcher is a static member variable of WeatherProvider and is a 
52        * common convention in Android programming. 
53        */ 
54       private static final UriMatcher sUriMatcher = buildUriMatcher(); 
55       private WeatherDbHelper mOpenHelper; 
56    
57       /** 
58        * Creates the UriMatcher that will match each URI to the CODE_WEATHER and 
59        * CODE_WEATHER_WITH_DATE constants defined above. 
60        * <p> 
61        * It's possible you might be thinking, "Why create a UriMatcher when you can use regular 
62        * expressions instead? After all, we really just need to match some patterns, and we can 
63        * use regular expressions to do that right?" Because you're not crazy, that's why. 
64        * <p> 
65        * UriMatcher does all the hard work for you. You just have to tell it which code to match 
66        * with which URI, and it does the rest automagically. Remember, the best programmers try 
67        * to never reinvent the wheel. If there is a solution for a problem that exists and has 
68        * been tested and proven, you should almost always use it unless there is a compelling 
69        * reason not to. 
70        * 
71        * @return A UriMatcher that correctly matches the constants for CODE_WEATHER and CODE_WEATHER_WITH_DATE 
72        */ 
73       public static UriMatcher buildUriMatcher() { 
74    
75           /* 
76            * All paths added to the UriMatcher have a corresponding code to return when a match is 
77            * found. The code passed into the constructor of UriMatcher here represents the code to 
78            * return for the root URI. It's common to use NO_MATCH as the code for this case. 
79            */ 
80           final UriMatcher matcher = new UriMatcher(UriMatcher.NO_MATCH); 
81           final String authority = WeatherContract.CONTENT_AUTHORITY; 
82    
83           /* 
84            * For each type of URI you want to add, create a corresponding code. Preferably, these are 
85            * constant fields in your class so that you can use them throughout the class and you no 
86            * they aren't going to change. In Sunshine, we use CODE_WEATHER or CODE_WEATHER_WITH_DATE. 
87            */ 
88    
89           /* This URI is content://com.example.android.sunshine/weather/ */ 
90           matcher.addURI(authority, WeatherContract.PATH_WEATHER, CODE_WEATHER); 
91    
92           /* 
93            * This URI would look something like content://com.example.android.sunshine/weather/1472214172 
94            * The "/#" signifies to the UriMatcher that if PATH_WEATHER is followed by ANY number, 
95            * that it should return the CODE_WEATHER_WITH_DATE code 
96            */ 
97           matcher.addURI(authority, WeatherContract.PATH_WEATHER + "/#", CODE_WEATHER_WITH_DATE); 
98    
99           return matcher; 
100      }

102      /** 
103       * In onCreate, we initialize our content provider on startup. This method is called for all 
104       * registered content providers on the application main thread at application launch time. 
105       * It must not perform lengthy operations, or application startup will be delayed. 
106       * 
107       * Nontrivial initialization (such as opening, upgrading, and scanning 
108       * databases) should be deferred until the content provider is used (via {@link #query}, 
109       * {@link #bulkInsert(Uri, ContentValues[])}, etc). 
110       * 
111       * Deferred initialization keeps application startup fast, avoids unnecessary work if the 
112       * provider turns out not to be needed, and stops database errors (such as a full disk) from 
113       * halting application launch. 
114       * 
115       * @return true if the provider was successfully loaded, false otherwise 
116       */ 
117      @Override 
118      public boolean onCreate() { 
119          /* 
120           * As noted in the comment above, onCreate is run on the main thread, so performing any 
121           * lengthy operations will cause lag in your app. Since WeatherDbHelper's constructor is 
122           * very lightweight, we are safe to perform that initialization here. 
123           */ 
124          mOpenHelper = new WeatherDbHelper(getContext()); 
125          return true; 
126      }

129  //  COMPLETED (1) Implement the bulkInsert method 
130      /** 
131       * Handles requests to insert a set of new rows. In Sunshine, we are only going to be 
132       * inserting multiple rows of data at a time from a weather forecast. There is no use case 
133       * for inserting a single row of data into our ContentProvider, and so we are only going to 
134       * implement bulkInsert. In a normal ContentProvider's implementation, you will probably want 
135       * to provide proper functionality for the insert method as well. 
136       * 
137       * @param uri    The content:// URI of the insertion request. 
138       * @param values An array of sets of column_name/value pairs to add to the database. 
139       *               This must not be {@code null}. 
140       * 
141       * @return The number of values that were inserted. 
142       */ 
143      @Override 
144      public int bulkInsert(@NonNull Uri uri, @NonNull ContentValues[] values) { 
145          final SQLiteDatabase db = mOpenHelper.getWritableDatabase(); 
146   
147          switch (sUriMatcher.match(uri)) { 
148   
149  //          COMPLETED (2) Only perform our implementation of bulkInsert if the URI matches the CODE_WEATHER code 
150              case CODE_WEATHER: 
151                  db.beginTransaction(); 
152                  int rowsInserted = 0; 
153                  try { 
154                      for (ContentValues value : values) { 
155                          long weatherDate = 
156                                  value.getAsLong(WeatherContract.WeatherEntry.COLUMN_DATE); 
157                          if (!SunshineDateUtils.isDateNormalized(weatherDate)) { 
158                              throw new IllegalArgumentException("Date must be normalized to insert"); 
159                          } 
160   
161                          long _id = db.insert(WeatherContract.WeatherEntry.TABLE_NAME, null, value); 
162                          if (_id != -1) { 
163                              rowsInserted++; 
164                          } 
165                      } 
166                      db.setTransactionSuccessful(); 
167                  } finally { 
168                      db.endTransaction(); 
169                  } 
170   
171                  if (rowsInserted > 0) { 
172                      getContext().getContentResolver().notifyChange(uri, null); 
173                  } 
174   
175  //              COMPLETED (3) Return the number of rows inserted from our implementation of bulkInsert 
176                  return rowsInserted; 
177   
178  //          COMPLETED (4) If the URI does match match CODE_WEATHER, return the super implementation of bulkInsert 
179              default: 
180                  return super.bulkInsert(uri, values); 
181          } 
182      }

184      /** 
185       * Handles query requests from clients. We will use this method in Sunshine to query for all 
186       * of our weather data as well as to query for the weather on a particular day. 
187       * 
188       * @param uri           The URI to query 
189       * @param projection    The list of columns to put into the cursor. If null, all columns are 
190       *                      included. 
191       * @param selection     A selection criteria to apply when filtering rows. If null, then all 
192       *                      rows are included. 
193       * @param selectionArgs You may include ?s in selection, which will be replaced by 
194       *                      the values from selectionArgs, in order that they appear in the 
195       *                      selection. 
196       * @param sortOrder     How the rows in the cursor should be sorted. 
197       * @return A Cursor containing the results of the query. In our implementation, 
198       */ 
199      @Override 
200      public Cursor query(@NonNull Uri uri, String[] projection, String selection, 
201                          String[] selectionArgs, String sortOrder) { 
202   
203          Cursor cursor; 
204   
205          /* 
206           * Here's the switch statement that, given a URI, will determine what kind of request is 
207           * being made and query the database accordingly. 
208           */ 
209          switch (sUriMatcher.match(uri)) { 
210   
211              /* 
212               * When sUriMatcher's match method is called with a URI that looks something like this 
213               * 
214               *      content://com.example.android.sunshine/weather/1472214172 
215               * 
216               * sUriMatcher's match method will return the code that indicates to us that we need 
217               * to return the weather for a particular date. The date in this code is encoded in 
218               * milliseconds and is at the very end of the URI (1472214172) and can be accessed 
219               * programmatically using Uri's getLastPathSegment method. 
220               * 
221               * In this case, we want to return a cursor that contains one row of weather data for 
222               * a particular date. 
223               */ 
224              case CODE_WEATHER_WITH_DATE: { 
225   
226                  /* 
227                   * In order to determine the date associated with this URI, we look at the last 
228                   * path segment. In the comment above, the last path segment is 1472214172 and 
229                   * represents the number of seconds since the epoch, or UTC time. 
230                   */ 
231                  String normalizedUtcDateString = uri.getLastPathSegment(); 
232   
233                  /* 
234                   * The query method accepts a string array of arguments, as there may be more 
235                   * than one "?" in the selection statement. Even though in our case, we only have 
236                   * one "?", we have to create a string array that only contains one element 
237                   * because this method signature accepts a string array. 
238                   */ 
239                  String[] selectionArguments = new String[]{normalizedUtcDateString}; 
240   
241                  cursor = mOpenHelper.getReadableDatabase().query( 
242                          /* Table we are going to query */ 
243                          WeatherContract.WeatherEntry.TABLE_NAME, 
244                          /* 
245                           * A projection designates the columns we want returned in our Cursor. 
246                           * Passing null will return all columns of data within the Cursor. 
247                           * However, if you don't need all the data from the table, it's best 
248                           * practice to limit the columns returned in the Cursor with a projection. 
249                           */ 
250                          projection, 
251                          /* 
252                           * The URI that matches CODE_WEATHER_WITH_DATE contains a date at the end 
253                           * of it. We extract that date and use it with these next two lines to 
254                           * specify the row of weather we want returned in the cursor. We use a 
255                           * question mark here and then designate selectionArguments as the next 
256                           * argument for performance reasons. Whatever Strings are contained 
257                           * within the selectionArguments array will be inserted into the 
258                           * selection statement by SQLite under the hood. 
259                           */ 
260                          WeatherContract.WeatherEntry.COLUMN_DATE + " = ? ", 
261                          selectionArguments, 
262                          null, 
263                          null, 
264                          sortOrder); 
265   
266                  break; 
267              } 
268   
269              /* 
270               * When sUriMatcher's match method is called with a URI that looks EXACTLY like this 
271               * 
272               *      content://com.example.android.sunshine/weather/ 
273               * 
274               * sUriMatcher's match method will return the code that indicates to us that we need 
275               * to return all of the weather in our weather table. 
276               * 
277               * In this case, we want to return a cursor that contains every row of weather data 
278               * in our weather table. 
279               */ 
280              case CODE_WEATHER: { 
281                  cursor = mOpenHelper.getReadableDatabase().query( 
282                          WeatherContract.WeatherEntry.TABLE_NAME, 
283                          projection, 
284                          selection, 
285                          selectionArgs, 
286                          null, 
287                          null, 
288                          sortOrder); 
289   
290                  break; 
291              } 
292   
293              default: 
294                  throw new UnsupportedOperationException("Unknown uri: " + uri); 
295          } 
296   
297          cursor.setNotificationUri(getContext().getContentResolver(), uri); 
298          return cursor; 
299      }
