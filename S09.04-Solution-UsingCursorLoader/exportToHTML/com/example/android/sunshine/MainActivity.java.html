179      /** 
180       * Uses the URI scheme for showing a location found on a map in conjunction with 
181       * an implicit Intent. This super-handy Intent is detailed in the "Common Intents" page of 
182       * Android's developer site: 
183       * 
184       * @see "http://developer.android.com/guide/components/intents-common.html#Maps" 
185       * <p> 
186       * Protip: Hold Command on Mac or Control on Windows and click that link to automagically 
187       * open the Common Intents page 
188       */ 
189      private void openPreferredLocationInMap() { 
190          double[] coords = SunshinePreferences.getLocationCoordinates(this); 
191          String posLat = Double.toString(coords[0]); 
192          String posLong = Double.toString(coords[1]); 
193          Uri geoLocation = Uri.parse("geo:" + posLat + "," + posLong); 
194   
195          Intent intent = new Intent(Intent.ACTION_VIEW); 
196          intent.setData(geoLocation); 
197   
198          if (intent.resolveActivity(getPackageManager()) != null) { 
199              startActivity(intent); 
200          } else { 
201              Log.d(TAG, "Couldn't call " + geoLocation.toString() + ", no receiving apps installed!"); 
202          } 
203      }

206      /** 
207       * Called by the {@link android.support.v4.app.LoaderManagerImpl} when a new Loader needs to be 
208       * created. This Activity only uses one loader, so we don't necessarily NEED to check the 
209       * loaderId, but this is certainly best practice. 
210       * 
211       * @param loaderId The loader ID for which we need to create a loader 
212       * @param bundle   Any arguments supplied by the caller 
213       * @return A new Loader instance that is ready to start loading. 
214       */ 
215      @Override 
216      public Loader<Cursor> onCreateLoader(int loaderId, Bundle bundle) { 
217   
218  //      COMPLETED (23) Remove the onStartLoading method declaration 
219  //      COMPLETED (24) Remove the loadInBackground method declaration 
220  //      COMPLETED (25) Remove the deliverResult method declaration 
221   
222          switch (loaderId) { 
223   
224  //          COMPLETED (22) If the loader requested is our forecast loader, return the appropriate CursorLoader 
225              case ID_FORECAST_LOADER: 
226                  /* URI for all rows of weather data in our weather table */ 
227                  Uri forecastQueryUri = WeatherContract.WeatherEntry.CONTENT_URI; 
228                  /* Sort order: Ascending by date */ 
229                  String sortOrder = WeatherContract.WeatherEntry.COLUMN_DATE + " ASC"; 
230                  /* 
231                   * A SELECTION in SQL declares which rows you'd like to return. In our case, we 
232                   * want all weather data from today onwards that is stored in our weather table. 
233                   * We created a handy method to do that in our WeatherEntry class. 
234                   */ 
235                  String selection = WeatherContract.WeatherEntry.getSqlSelectForTodayOnwards(); 
236   
237                  return new CursorLoader(this, 
238                          forecastQueryUri, 
239                          MAIN_FORECAST_PROJECTION, 
240                          selection, 
241                          null, 
242                          sortOrder); 
243   
244              default: 
245                  throw new RuntimeException("Loader Not Implemented: " + loaderId); 
246          } 
247      }
