30   /** 
31    * {@link ForecastAdapter} exposes a list of weather forecasts 
32    * from a {@link android.database.Cursor} to a {@link android.support.v7.widget.RecyclerView}. 
33    */ 
34   class ForecastAdapter extends RecyclerView.Adapter<ForecastAdapter.ForecastAdapterViewHolder> { 
35    
36   //  COMPLETED (14) Remove the mWeatherData declaration 
37    
38   //  COMPLETED (1) Declare a private final Context field called mContext 
39       /* The context we use to utility methods, app resources and layout inflaters */ 
40       private final Context mContext; 
41    
42       /* 
43        * Below, we've defined an interface to handle clicks on items within this Adapter. In the 
44        * constructor of our ForecastAdapter, we receive an instance of a class that has implemented 
45        * said interface. We store that instance in this variable to call the onClick method whenever 
46        * an item is clicked in the list. 
47        */ 
48       final private ForecastAdapterOnClickHandler mClickHandler; 
49    
50       /** 
51        * The interface that receives onClick messages. 
52        */ 
53       public interface ForecastAdapterOnClickHandler { 
54           void onClick(String weatherForDay); 
55       } 
56    
57   //  COMPLETED (2) Declare a private Cursor field called mCursor 
58       private Cursor mCursor; 
59    
60   //  COMPLETED (3) Add a Context field to the constructor and store that context in mContext 
61       /** 
62        * Creates a ForecastAdapter. 
63        * 
64        * @param context Used to talk to the UI and app resources 
65        * @param clickHandler The on-click handler for this adapter. This single handler is called 
66        *                     when an item is clicked. 
67        */ 
68       public ForecastAdapter(@NonNull Context context, ForecastAdapterOnClickHandler clickHandler) { 
69           mContext = context; 
70           mClickHandler = clickHandler; 
71       } 
72    
73       /** 
74        * This gets called when each new ViewHolder is created. This happens when the RecyclerView 
75        * is laid out. Enough ViewHolders will be created to fill the screen and allow for scrolling. 
76        * 
77        * @param viewGroup The ViewGroup that these ViewHolders are contained within. 
78        * @param viewType  If your RecyclerView has more than one type of item (like ours does) you 
79        *                  can use this viewType integer to provide a different layout. See 
80        *                  {@link android.support.v7.widget.RecyclerView.Adapter#getItemViewType(int)} 
81        *                  for more details. 
82        * @return A new ForecastAdapterViewHolder that holds the View for each list item 
83        */ 
84       @Override 
85       public ForecastAdapterViewHolder onCreateViewHolder(ViewGroup viewGroup, int viewType) { 
86    
87           View view = LayoutInflater 
88                   .from(mContext) 
89                   .inflate(R.layout.forecast_list_item, viewGroup, false); 
90    
91           view.setFocusable(true); 
92    
93           return new ForecastAdapterViewHolder(view); 
94       } 
95    
96       /** 
97        * OnBindViewHolder is called by the RecyclerView to display the data at the specified 
98        * position. In this method, we update the contents of the ViewHolder to display the weather 
99        * details for this particular position, using the "position" argument that is conveniently 
100       * passed into us. 
101       * 
102       * @param forecastAdapterViewHolder The ViewHolder which should be updated to represent the 
103       *                                  contents of the item at the given position in the data set. 
104       * @param position                  The position of the item within the adapter's data set. 
105       */ 
106      @Override 
107      public void onBindViewHolder(ForecastAdapterViewHolder forecastAdapterViewHolder, int position) { 
108  //      COMPLETED (5) Delete the current body of onBindViewHolder 
109   
110  //      COMPLETED (6) Move the cursor to the appropriate position 
111          mCursor.moveToPosition(position); 
112   
113   
114          /******************* 
115           * Weather Summary * 
116           *******************/ 
117  //      COMPLETED (7) Generate a weather summary with the date, description, high and low 
118          /* Read date from the cursor */ 
119          long dateInMillis = mCursor.getLong(MainActivity.INDEX_WEATHER_DATE); 
120          /* Get human readable string using our utility method */ 
121          String dateString = SunshineDateUtils.getFriendlyDateString(mContext, dateInMillis, false); 
122          /* Use the weatherId to obtain the proper description */ 
123          int weatherId = mCursor.getInt(MainActivity.INDEX_WEATHER_CONDITION_ID); 
124          String description = SunshineWeatherUtils.getStringForWeatherCondition(mContext, weatherId); 
125          /* Read high temperature from the cursor (in degrees celsius) */ 
126          double highInCelsius = mCursor.getDouble(MainActivity.INDEX_WEATHER_MAX_TEMP); 
127          /* Read low temperature from the cursor (in degrees celsius) */ 
128          double lowInCelsius = mCursor.getDouble(MainActivity.INDEX_WEATHER_MIN_TEMP); 
129   
130          String highAndLowTemperature = 
131                  SunshineWeatherUtils.formatHighLows(mContext, highInCelsius, lowInCelsius); 
132   
133          String weatherSummary = dateString + " - " + description + " - " + highAndLowTemperature; 
134   
135  //      COMPLETED (8) Display the summary that you created above 
136          forecastAdapterViewHolder.weatherSummary.setText(weatherSummary); 
137      } 
138   
139      /** 
140       * This method simply returns the number of items to display. It is used behind the scenes 
141       * to help layout our Views and for animations. 
142       * 
143       * @return The number of items available in our forecast 
144       */ 
145      @Override 
146      public int getItemCount() { 
147  //      COMPLETED (9) Delete the current body of getItemCount 
148  //      COMPLETED (10) If mCursor is null, return 0. Otherwise, return the count of mCursor 
149          if (null == mCursor) return 0; 
150          return mCursor.getCount(); 
151      } 
152   
153  //  COMPLETED (11) Create a new method that allows you to swap Cursors. 
154      /** 
155       * Swaps the cursor used by the ForecastAdapter for its weather data. This method is called by 
156       * MainActivity after a load has finished, as well as when the Loader responsible for loading 
157       * the weather data is reset. When this method is called, we assume we have a completely new 
158       * set of data, so we call notifyDataSetChanged to tell the RecyclerView to update. 
159       * 
160       * @param newCursor the new cursor to use as ForecastAdapter's data source 
161       */ 
162      void swapCursor(Cursor newCursor) { 
163          mCursor = newCursor; 
164  //      COMPLETED (12) After the new Cursor is set, call notifyDataSetChanged 
165          notifyDataSetChanged(); 
166      } 
167   
168      /** 
169       * A ViewHolder is a required part of the pattern for RecyclerViews. It mostly behaves as 
170       * a cache of the child views for a forecast item. It's also a convenient place to set an 
171       * OnClickListener, since it has access to the adapter and the views. 
172       */ 
173      class ForecastAdapterViewHolder extends RecyclerView.ViewHolder implements View.OnClickListener { 
174          final TextView weatherSummary; 
175   
176          ForecastAdapterViewHolder(View view) { 
177              super(view); 
178   
179              weatherSummary = (TextView) view.findViewById(R.id.tv_weather_data); 
180   
181              view.setOnClickListener(this); 
182          } 
183   
184          /** 
185           * This gets called by the child views during a click. We fetch the date that has been 
186           * selected, and then call the onClick handler registered with this adapter, passing that 
187           * date. 
188           * 
189           * @param v the View that was clicked 
190           */ 
191          @Override 
192          public void onClick(View v) { 
193              //  COMPLETED (13) Instead of passing the String from the data array, use the weatherSummary text! 
194              String weatherForDay = weatherSummary.getText().toString(); 
195              mClickHandler.onClick(weatherForDay); 
196          } 
197      }
