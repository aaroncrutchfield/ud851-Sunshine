84       /** 
85        * Returns the location currently set in Preferences. The default location this method 
86        * will return is "94043,USA", which is Mountain View, California. Mountain View is the 
87        * home of the headquarters of the Googleplex! 
88        * 
89        * @param context Context used to get the SharedPreferences 
90        * @return Location The current user has set in SharedPreferences. Will default to 
91        * "94043,USA" if SharedPreferences have not been implemented yet. 
92        */ 
93       public static String getPreferredWeatherLocation(Context context) { 
94           // COMPLETED (1) Return the user's preferred location 
95           SharedPreferences prefs = PreferenceManager 
96                   .getDefaultSharedPreferences(context); 
97           String keyForLocation = context.getString(R.string.pref_location_key); 
98           String defaultLocation = context.getString(R.string.pref_location_default); 
99           return prefs.getString(keyForLocation, defaultLocation); 
100      } 
101   
102      /** 
103       * Returns true if the user has selected metric temperature display. 
104       * 
105       * @param context Context used to get the SharedPreferences 
106       * 
107       * @return true If metric display should be used 
108       */ 
109      public static boolean isMetric(Context context) { 
110          // COMPLETED (2) Return true if the user's preference for units is metric, false otherwise 
111          SharedPreferences prefs = PreferenceManager 
112                  .getDefaultSharedPreferences(context); 
113          String keyForUnits = context.getString(R.string.pref_units_key); 
114          String defaultUnits = context.getString(R.string.pref_units_metric); 
115          String preferredUnits = prefs.getString(keyForUnits, defaultUnits); 
116          String metric = context.getString(R.string.pref_units_metric); 
117          boolean userPrefersMetric; 
118          if (metric.equals(preferredUnits)) { 
119              userPrefersMetric = true; 
120          } else { 
121              userPrefersMetric = false; 
122          } 
123          return userPrefersMetric; 
124      }
