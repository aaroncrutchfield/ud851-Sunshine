16   package com.example.android.sunshine; 
17    
18   import android.content.Context; 
19   import android.support.v7.widget.RecyclerView; 
20   import android.view.LayoutInflater; 
21   import android.view.View; 
22   import android.view.View.OnClickListener; 
23   import android.view.ViewGroup; 
24   import android.widget.TextView; 
25    
26   /** 
27    * {@link ForecastAdapter} exposes a list of weather forecasts to a 
28    * {@link android.support.v7.widget.RecyclerView} 
29    */ 
30   public class ForecastAdapter extends RecyclerView.Adapter<ForecastAdapter.ForecastAdapterViewHolder> { 
31    
32       private String[] mWeatherData; 
33    
34       // COMPLETED (3) Create a final private ForecastAdapterOnClickHandler called mClickHandler 
35       /* 
36        * An on-click handler that we've defined to make it easy for an Activity to interface with 
37        * our RecyclerView 
38        */ 
39       private final ForecastAdapterOnClickHandler mClickHandler; 
40    
41       // COMPLETED (1) Add an interface called ForecastAdapterOnClickHandler 
42       // COMPLETED (2) Within that interface, define a void method that access a String as a parameter 
43       /** 
44        * The interface that receives onClick messages. 
45        */ 
46       public interface ForecastAdapterOnClickHandler { 
47           void onClick(String weatherForDay); 
48       } 
49    
50       // COMPLETED (4) Add a ForecastAdapterOnClickHandler as a parameter to the constructor and store it in mClickHandler 
51       /** 
52        * Creates a ForecastAdapter. 
53        * 
54        * @param clickHandler The on-click handler for this adapter. This single handler is called 
55        *                     when an item is clicked. 
56        */ 
57       public ForecastAdapter(ForecastAdapterOnClickHandler clickHandler) { 
58           mClickHandler = clickHandler; 
59       } 
60    
61       // COMPLETED (5) Implement OnClickListener in the ForecastAdapterViewHolder class 
62       /** 
63        * Cache of the children views for a forecast list item. 
64        */ 
65       public class ForecastAdapterViewHolder extends RecyclerView.ViewHolder implements OnClickListener { 
66           public final TextView mWeatherTextView; 
67    
68           public ForecastAdapterViewHolder(View view) { 
69               super(view); 
70               mWeatherTextView = (TextView) view.findViewById(R.id.tv_weather_data); 
71               // COMPLETED (7) Call setOnClickListener on the view passed into the constructor (use 'this' as the OnClickListener) 
72               view.setOnClickListener(this); 
73           } 
74    
75           // COMPLETED (6) Override onClick, passing the clicked day's data to mClickHandler via its onClick method 
76           /** 
77            * This gets called by the child views during a click. 
78            * 
79            * @param v The View that was clicked 
80            */ 
81           @Override 
82           public void onClick(View v) { 
83               int adapterPosition = getAdapterPosition(); 
84               String weatherForDay = mWeatherData[adapterPosition]; 
85               mClickHandler.onClick(weatherForDay); 
86           } 
87       } 
88    
89       /** 
90        * This gets called when each new ViewHolder is created. This happens when the RecyclerView 
91        * is laid out. Enough ViewHolders will be created to fill the screen and allow for scrolling. 
92        * 
93        * @param viewGroup The ViewGroup that these ViewHolders are contained within. 
94        * @param viewType  If your RecyclerView has more than one type of item (which ours doesn't) you 
95        *                  can use this viewType integer to provide a different layout. See 
96        *                  {@link android.support.v7.widget.RecyclerView.Adapter#getItemViewType(int)} 
97        *                  for more details. 
98        * @return A new ForecastAdapterViewHolder that holds the View for each list item 
99        */ 
100      @Override 
101      public ForecastAdapterViewHolder onCreateViewHolder(ViewGroup viewGroup, int viewType) { 
102          Context context = viewGroup.getContext(); 
103          int layoutIdForListItem = R.layout.forecast_list_item; 
104          LayoutInflater inflater = LayoutInflater.from(context); 
105          boolean shouldAttachToParentImmediately = false; 
106   
107          View view = inflater.inflate(layoutIdForListItem, viewGroup, shouldAttachToParentImmediately); 
108          return new ForecastAdapterViewHolder(view); 
109      } 
110   
111      /** 
112       * OnBindViewHolder is called by the RecyclerView to display the data at the specified 
113       * position. In this method, we update the contents of the ViewHolder to display the weather 
114       * details for this particular position, using the "position" argument that is conveniently 
115       * passed into us. 
116       * 
117       * @param forecastAdapterViewHolder The ViewHolder which should be updated to represent the 
118       *                                  contents of the item at the given position in the data set. 
119       * @param position                  The position of the item within the adapter's data set. 
120       */ 
121      @Override 
122      public void onBindViewHolder(ForecastAdapterViewHolder forecastAdapterViewHolder, int position) { 
123          String weatherForThisDay = mWeatherData[position]; 
124          forecastAdapterViewHolder.mWeatherTextView.setText(weatherForThisDay); 
125      } 
126   
127      /** 
128       * This method simply returns the number of items to display. It is used behind the scenes 
129       * to help layout our Views and for animations. 
130       * 
131       * @return The number of items available in our forecast 
132       */ 
133      @Override 
134      public int getItemCount() { 
135          if (null == mWeatherData) return 0; 
136          return mWeatherData.length; 
137      } 
138   
139      /** 
140       * This method is used to set the weather forecast on a ForecastAdapter if we've already 
141       * created one. This is handy when we get new data from the web but don't want to create a 
142       * new ForecastAdapter to display it. 
143       * 
144       * @param weatherData The new weather data to be displayed. 
145       */ 
146      public void setWeatherData(String[] weatherData) { 
147          mWeatherData = weatherData; 
148          notifyDataSetChanged(); 
149      } 
150  }
